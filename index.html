Modules

Основные фичи модулей это функции export и import

// calculator/lib/calc.js
export function square(x) {
  return x * x;
}
export const MY_CONSTANT = 123;
// calculator/main.js
import { square } from 'lib/calc';

Также можно импортировать модуль как объект, значениями свойств которого будут экспортированные значения:

import 'lib/calc' as c;

Есди модуль экспортирует только одно значение (большой класс, например). В таком случае удобно определить это значение как экспортируемое по умолчанию

// calculator/lib/calc.js
export default class {
  constructor(shape) {
    this.shape = shape;
  }
};

В таком случае при импорте модуля можно опускать фигурные скобки

// calculator/main.js
import Calc  from 'lib/calc';
let c = new Calc(10);

Также можно переименовывать значения во время экспорта

export { square as squ, MY_CONSTANT as SOME_CONSTANT };

Spread operator and rest parameters

Параметры по умолчанию позволяют задавать формальным параметрам функции значения по умолчанию, если для них не указано значение или передан undefined.

function multiply(a, b = 1) {
  return a * b;
}

multiply(5); // 5

Rest parameters

function f (x, y, ...a) {
    console.log(a); //["hello", true, 7]
    return (x + y) * a.length;
}
f(1, 2, "hello", true, 7); //6


Использование spread operator'а

let parts = ['первый', 'второй'];
let lyrics = ['третий', ...parts, 'четвертый'];

// ['третий', 'первый', 'второй', 'четвертый']

Копирование массива

let arr = [1, 2, 3, 4, 5];
let data = [...arr];
console.log(data); // [1, 2, 3, 4, 5]

Из массива в значения


let obj = [1,3,4,7];
let arr = [3, 7];
console.log(_.without(obj, ...arr)); //[1, 4]

Destructuring(Разбор)

Разбор массивов

let array = ["one", "two", "three"];
// без деструктурирования
let one   = foo[0];
let two   = foo[1];
let three = foo[2];
// с деструктурированием
let [one, two, three] = array;

Разбор объектов

let object = {p: 42, q: true};
let {p, q} = object;
console.log(p); // 42
console.log(q); // true
// Объявление новых переменных
let {p: foo, q: bar} = object;
console.log(foo); // 42
console.log(bar); // true

Swap значений

let a = 1, b = 2;
[b, a] = [a, b];
console.log(a, b);
// => 2 1

Разбор сложных структур

let {prop: x, prop2: [, y]} = {prop: 5, prop2: [10, 100]};
console.log(x, y);// 5 100

Ох уж эти Promise'ы

В двух словах Promise — это объект, обладающий методами .then() and .catch()

var promise = new Promise(function(resolve, reject) {
  // здесь вытворяй что угодно, если хочешь асинхронно, потом…
  if (/* ..если всё закончилось успехом */) {
    resolve("Работает!");
  }
  else {
    reject(Error("Сломалось"));
  }
});

Так как методы Promise.prototype.then и Promise.prototype.catch сами возвращают обещания, их можно вызывать цепочкой, создавая соединения. 

let promise = new Promise(function(resolve, reject) {
  resolve(1);
});
promise.then(function(val) {
 return val + 2;
}).then(function(val) {
  console.log(val); // 3
});

Обработка нескольких Promise'ов

let p1 = new Promise(...),
p2 = new Promise(...)
p3 = Promise.all([p1, p2]);
p3.then(...).catch(...);

Подводим итоги:

Плюсы использования ES6

  • Это модно и современно, всё больше литературы использует ES6 и к этому надо привыкать
  • Это удобно, необходимо написать меньше строк для получения аналогичного результата

Минусы использования ES6

  • Не все фичи поддерживаются браузерами, поэтому приходится прибегать к помощи transpiler'ов
  • При конвертации ES6 в ES5 меняется код на выходе

Что дальше?

Ну а дальше ES7, а точнее ES2016. Теперь решено выпускать релизы значительно чаще (раз в год), и изменения могут вносится постепенно. Новые возможности языка включаются в релиз только после того, как они полностью готовы и есть минимум две имплементации, которые прошли достаточное тестирование в процессе практического применения.
