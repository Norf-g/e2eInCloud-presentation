let, const

function varTest() {
  var x = 31;
  if (true) {
    var x = 71;  // та же переменная!
    console.log(x);  // 71
  }
  console.log(x);  // 71
}

function letTest() {
  let x = 31;
  if (true) {
    let x = 71;  // другая переменная
    console.log(x);  // 71
  }
  console.log(x);  // 31
}

Map

Объект Map является простым ассоциативным массивом. Любые типы (как объекты, так и примитивы) могут быть использованы в качестве ключа или значения.

Свойства экземпляров класса map:

Map.prototype.size

Методы экземпляров класса map:

Map.prototype.clear(); .delete(key); .entries(); .forEach(callbackFn[, thisArg]); .get(key); .has(key); .keys(); .set(key, value); .values();

Set

Объекты Set представляют коллекции значений, по который вы можете выполнить обход в порядке вставки элементов. Значение элемента в Set может присутствовать только в одном экземпляре, что обеспечивает его уникальность в рамках коллекции Set.

Classes

Объявление класса

//В отличии от функции при объявлении класса
//не происходит подъем(hoisted) 			
class Polygon {
  constructor(height, width) {
    this.height = height;
    this.width = width;
  }
  get area() {
    return this.calcArea();
  }
  //Статические методы вызываются без инстанцирования класса и 
  //не могут быть вызваны у экземпляров класса. 
  static distance(a, b) {
     const dx = a.x - b.x;
     const dy = a.y - b.y;

     return Math.sqrt(dx*dx + dy*dy);
   }
}

Spread operator and rest parameters

Параметры по умолчанию позволяют задавать формальным параметрам функции значения по умолчанию, если для них не указано значение или передан undefined.

function multiply(a, b = 1) {
  return a*b;
}

multiply(5); // 5

Rest parameters

function f (x, y, ...a) {
    console.log(a); //["hello", true, 7]
    return (x + y) * a.length;
}
f(1, 2, "hello", true, 7);
